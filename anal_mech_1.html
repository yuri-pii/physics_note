\begin{document} \section*{解析力学まとめノート（本質中心・定義と単位付き）} \begin{boxenv} \section*{0. 視点の要約} 解析力学とは、運動を \textbf{関数空間における停留問題} として記述する理論である。 「最小化」ではなく、 \textbf{一次変化が消える（停留）} ことが本質である。 \end{boxenv} \begin{boxenv} \section*{1. 基本対象と定義} \subsection*{一般化座標と時間} \begin{itemize} \item 時間：\( t \) \quad [s] \item 一般化座標：\( q_i(t)\ (i=1,\dots,n) \) \item 一般化速度： \[ \dot{q}_i = \frac{dq_i}{dt} \quad [q_i/\mathrm{s}] \] \end{itemize} \subsection*{ラグランジアン} \begin{eqenv} \[ L(q,\dot{q},t)=T(q,\dot{q},t)-V(q,t) \] \end{eqenv} {\deftext \begin{itemize} \item \(T\)：運動エネルギー [J]（速度に依存） \item \(V\)：ポテンシャルエネルギー [J]（位置に依存） \item 意味： 運動エネルギーは「動こうとする傾向」， ポテンシャルは「配置による制約」を表す。 その差 \(T-V\) を用いることで， 実際に実現する運動が作用の停留条件として定まる。 \end{itemize} } \subsection*{作用（Action）} \begin{eqenv} \[ S[q] = \int_{t_1}^{t_2} L(q,\dot{q},t)\,dt \] \end{eqenv} {\deftext \begin{itemize} \item 経路全体の評価量 \item 単位：[J\,s] \end{itemize} } \end{boxenv} \begin{boxenv} \section*{2. 微分と変分} \subsection*{時間微分} \begin{eqenv} \[ \dot{q}_i(t)=\frac{dq_i(t)}{dt} \] \end{eqenv} {\deftext 同一の経路上での時間方向の変化を見る操作。 } \subsection*{変分} \begin{eqenv} \[ q(t)\rightarrow q(t)+\varepsilon\eta(t) \] \[ \delta q(t)= \left. \frac{d}{d\varepsilon} \bigl[q(t)+\varepsilon\eta(t)\bigr] \right|_{\varepsilon=0} =\eta(t) \] \end{eqenv} {\deftext \begin{itemize} \item \( \varepsilon \)：無次元微小量 \item \( \eta(t) \)：端点で0となる任意関数 \end{itemize} } \subsection*{交換則} \begin{eqenv} \[ \delta\dot{q}=\frac{d}{dt}(\delta q) \] \end{eqenv} \end{boxenv} \begin{boxenv} \section*{3. 停留条件} \begin{eqenv} \[ \delta S=0 \] \end{eqenv} {\deftext 任意の変分方向に対して一次変化が消える。 最小・最大・鞍点を含む。 } \end{boxenv} \begin{boxenv} \section*{4. オイラー＝ラグランジュ方程式} \begin{eqenv} \[ \frac{d}{dt} \left( \frac{\partial L}{\partial\dot{q}_i} \right) - \frac{\partial L}{\partial q_i} =0 \] \end{eqenv} {\deftext \( \partial L/\partial\dot{q}_i \) は共役運動量に対応する。 } \end{boxenv} \begin{boxenv} \section*{5. 共役運動量とエネルギー} \subsection*{共役運動量} \begin{eqenv} \[ p_i=\frac{\partial L}{\partial\dot{q}_i} \] \end{eqenv} \subsection*{エネルギー} \begin{eqenv} \[ E=\sum_i \dot{q}_i p_i - L \] \end{eqenv} \end{boxenv} \begin{boxenv} \section*{6. 拘束と未定乗数} \begin{eqenv} \[ f(q,t)=0 \] \end{eqenv} \begin{eqenv} \[ \tilde{L}=L+\lambda(t)f(q,t) \] \end{eqenv} \end{boxenv} \begin{boxenv} \section*{7. 散逸} \begin{eqenv} \[ R=\frac{1}{2}\sum_i\gamma_i\dot{q}_i^2 \] \end{eqenv} \begin{eqenv} \[ \frac{d}{dt} \left( \frac{\partial L}{\partial\dot{q}_i} \right) - \frac{\partial L}{\partial q_i} = - \frac{\partial R}{\partial\dot{q}_i} \] \end{eqenv} \end{boxenv} \begin{boxenv} \section*{8. ハミルトン形式} \begin{eqenv} \[ H(q,p,t)=\sum_i p_i\dot{q}_i-L \] \end{eqenv} \begin{eqenv} \[ \dot{q}_i=\frac{\partial H}{\partial p_i}, \qquad \dot{p}_i=-\frac{\partial H}{\partial q_i} \] \end{eqenv} \end{boxenv} \begin{boxenv} \section*{9. 量子論との接続} \begin{eqenv} \[ \text{重み}\sim e^{iS/\hbar} \] \end{eqenv} \end{boxenv} \vspace{1em} {\small このノートは「定義 → 構造 → 必然性」を忘れないための参照用まとめである。 } \end{document}
